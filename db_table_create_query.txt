CREATE DATABASE first_rest;

USE first_rest;

CREATE TABLE users (
id int primary key auto_increment,
user_id varchar(30) unique,
password varchar(255),
nickname varchar(30)
);

CREATE TABLE board (
id int primary key auto_increment,
users_id int,
title varchar(255),
content text
);

CREATE TABLE token (
id int primary key auto_increment,
users_id int,
token varchar(255),
exp int
);
